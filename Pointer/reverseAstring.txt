#include<bits/stdc++.h>
using namespace std;

void reverseAstring(char *arr,int n)
{
    char *start=arr;
    char *end=arr+n-1;
    while(start<end)
    {
        char temp=*start;
        *start=*end;
        *end=temp;
        start++;
        end--;
    }
    cout<<arr;
    
}



int main()
{
    char arr[]="hello";
    int n=strlen(arr);
    reverseAstring(arr,n);
}
