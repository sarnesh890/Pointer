#include<bits/stdc++.h>
using namespace std;

void findmax(int *arr,int n)
{
    int max=*arr;
    for(int i=0;i<n;i++){
        if(*(arr+i)>max)
        {
            max=*(arr+i);
        }
    }
    cout<<max;
}



int main()
{
    int arr[]={1,2,3,4,5};
    int n=sizeof(arr)/sizeof(arr[0]);
    findmax(arr,n);
}
